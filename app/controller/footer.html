The frontend controller:

require_once __DIR__ . "/vendor/autoload.php";

use JLaso\SlimRoutingManager\RoutingCacheManager;

$app = new \Slim\Slim();

new RoutingCacheManager(
    array(
        'cache'      => __DIR__ . '/cache/routing',
        'controller' => __DIR__ . '/app/controller',
    )
);

$app->run();
